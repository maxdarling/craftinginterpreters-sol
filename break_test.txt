// single nested break
{
  var i = 0;
  while (i < 5) {
    i = i + 1;
    if (i == 2) {
      break;
    }
  }
  if (i == 2) {
    print "single nested break: passed";
  } else {
    print "single nested break: error";
  }
}

// nested break
{
  var x = 0;
  while (x < 5) {
    var y = 0;
    while (true) {
      y = y + 1;
      if (y >= 1) {
        break;
      }
      x = 100; // goal: never reach this
    }
    x = x + 1;
  }
  if (x == 5) {
    print "nested break: passed";
  } else {
    print "nested break: error";
  }
}
